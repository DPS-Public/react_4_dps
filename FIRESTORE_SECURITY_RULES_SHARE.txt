Firebase Firestore Security Rules - Share UI Canvas & API Canvas Feature
========================================================================

UPDATED RULES:
-----------

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // UI Canvas collection
    match /ui_canvas/{canvasId} {
      // Authenticated users can read their own or shared canvases
      allow read: if request.auth != null || 
                     resource.data.isShared == true;
      
      // Only owner can write/update
      allow write: if request.auth != null && 
                      resource.data.ownerId == request.auth.uid;
      
      // Create new canvas
      allow create: if request.auth != null && 
                       request.resource.data.ownerId == request.auth.uid;
    }

    // API Canvas collection
    match /api_canvas/{canvasId} {
      // Public read allowed when isShared == true
      allow read: if request.auth != null || 
                     resource.data.isShared == true;
      
      // Only authenticated users can write
      allow write: if request.auth != null;
    }

    // UI Canvas Analysis collection
    match /ui_canvas_analysis/{canvasId} {
      // Authenticated users can read/write their own analyses
      allow read, write: if request.auth != null;
    }

    // API Canvas Analysis/History collections
    match /api_canvas_analysis/{canvasId} {
      // Authenticated users can read/write their own analyses
      allow read, write: if request.auth != null;
    }

    match /api_canvas_history/{canvasId} {
      // Authenticated users can read/write their own history
      allow read, write: if request.auth != null;
    }

    // All other collections - restrict to authenticated users
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
